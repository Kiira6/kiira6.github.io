Empezar a utilizar Lucifer

Para empezar a utilizar lucifer deberemos importarlo e inicializarlo.

main.cc
#include "./deps/Lucifer/src/core/lucifer.cc"

int main(int argc, char **argv)
{
  Lucifer::InitGame();

  return 0;
}

Una vez compilado lo ejecutamos y obtendremos la siguiente salida por consola

Ahora podemos crear nuestro primer módulo de prueba

exampleModule.cc
void exampleAwake()
{
  printf("Awake");
}

void exampleStart()
{
  printf("Start");
}

void exampleUpdate()
{
  printf("Update");
}

void exampleDestroy()
{
  printf("Destroy");
}

Lucifer::GameModule example_Module =
{
  &exampleAwake, &exampleStart, &exampleUpdate, &exampleDestroy
};

Y debemos añadir el fichero y cargar el modulo

main.cc
#include "./deps/Lucifer/src/core/lucifer.cc"
#include "./exampleModule.cc"

void Lucifer::LoadModules()
{
  AddModule(&example_Module);
}

int main(int argc, char **argv)
{
  Lucifer::InitGame();

  return 0;
}
